def produce(n=5, m=3, s=2):
    x = []
    y = []
    for i in range(-10,11):
        x.append(i)
        y.append(abs(i))


    return y, x
from matplotlib import pyplot as plt

data_y, data_x = produce(n=100,m=3,s=2)
plt.plot(data_x, data_y, color="red", marker="*")
plt.xlabel("Variable x")
plt.ylabel("Variable y")
plt.show()
