def get_l33ter(msg):
    letter_E ="e" or "E"
    Letter_I = "i" or "I"
    letter_O = "o" or "O"
    letter_A = "a" or "A"
    output = ""
    for letters in msg:
        if letters == letter_E:
            output += "3"
        elif letters == Letter_I:
            output += "1"
        elif letters == letter_O:
            output += "0"
        elif letters == letter_A:
            output += "4"
        else:
            output += letters
    return(output)
